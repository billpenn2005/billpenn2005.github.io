大水题🙃(本来想用线段树的)
原题

#include<iostream>
using namespace std;
int n,k,p;
int a[300001],b[300001];
int suma[300001];
int ans=0;
int main()
{
	cin>>n>>k>>p;
	for(int i=1;i<=n;++i)
	{
		cin>>a[i]>>b[i];
		suma[a[i]]++;
	}
	for(int i=1;i<=n;++i)
	{
		suma[a[i]]--;
		int minn=b[i];
		int tmpm=0;
		for(int j=i+1;j<=n;++j)
		{
			minn=min(minn,b[j]);
			if(a[j]==a[i])
			{
				if(minn<=p)
				{
					ans+=suma[a[i]]-tmpm;
					break;
				}
				else
				{
					tmpm++;
					continue;
				}
			}
		}
	}
	cout<<ans<<endl;
	return 0;
}
