#include<iostream>
using namespace std;
int f[11][11][11][11];
bool mark[11][11][11][11];
int a[11][11];
int n;
int dfs(int x1,int y1,int x2,int y2)
{
	//cout<<x1<<" "<<y1<<" "<<x2<<" "<<y2<<endl;
	if(mark[x1][y1][x2][y2]==1)
		return f[x1][y1][x2][y2];
	//cout<<1;
	if(x1==y1&&y1==x2&&x2==y2&&y2==1)
		return a[x1][y1];
	//cout<<2;
	if(x1<1||x2<1||y1<1||y2<1||x1>n||x2>n||y1>n||y2>n)
		return -(1<<30);
	//cout<<3;
	int ret=0;
	if(x1==x2&&y1==y2)
	{
		ret=a[x1][y1];
	}
	else
	{
		ret=a[x1][y1]+a[x2][y2];
	}
	int maxx=0;
	int xp[]={0,-1};
	int yp[]={-1,0};
	//cout<<5;
	for(int i=0;i<2;++i)
		for(int j=0;j<2;++j)
		{
		//	cout<<4;
			maxx=max(maxx,dfs(x1+xp[i],y1+yp[i],x2+xp[j],y2+yp[j]));
		}
	ret+=maxx;
	f[x1][y1][x2][y2]=ret;
	mark[x1][y1][x2][y2]=1;
	return ret;
}
int main()
{
	cin>>n;
	int x,y,z;
	while(cin>>x>>y>>z)
	{
		if(x==y&&y==z&&z==0)
			break;
		a[x][y]=z;
	}
	cout<<dfs(n,n,n,n);
	return 0;
}
